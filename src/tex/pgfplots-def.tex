\usepackage{pgfplots}
\usepackage{pgfplotstable}
\input{paultolcolors}

\pgfplotscreateplotcyclelist{TolVibList}{%
  {TolVibRed,     every mark/.append style={fill=TolVibRed},     mark=*},
  {TolVibBlue,    every mark/.append style={fill=TolVibBlue},    mark=*},
  {TolVibTeal,    every mark/.append style={fill=TolVibTeal},    mark=*},
  {TolVibCyan,    every mark/.append style={fill=TolVibCyan},    mark=*},
  {TolVibOrange,  every mark/.append style={fill=TolVibOrange},  mark=*},
  {TolVibMagenta, every mark/.append style={fill=TolVibMagenta}, mark=*},
  {TolGray,       every mark/.append style={fill=TolGray},       mark=*},
}

\pgfplotsset{
  compat=1.16,
  table/col sep=comma,
  width=14.7cm,
  height=5.5cm,
  /pgf/number format/1000 sep={},
  cycle list name=TolVibList,
  legend style={cells={anchor=west}},
  xtick pos=left,
  ytick pos=left,
  histogram/.style={%
    solid,
    const plot,
    no marks,
  },
  loghistogram/.style={%
    histogram,
    y filter/.code={%
      \ifx\pgfmathresult\empty%
        \def\pgfmathresult{-10}%
      \fi%
    },
  },
  loghistogramaxis/.style={%
    ymin=2e-1,
    /pgfplots/log basis=10,
    log origin=infty,
  },
  area legend/.append style={%
    legend image post style={draw opacity=0}
  },
}
