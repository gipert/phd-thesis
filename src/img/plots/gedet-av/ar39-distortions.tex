  width=5.6cm,
  xmin=41, xmax=300,
  cycle list name=ShadesList,
  no marks,
  legend style={%
    font=\footnotesize,
    draw=none,
    fill=none,
  },

  histogram/.append style={%
    table/x=energy,
    table/y expr=\thisrow{counts}/1e4,
  },
}

\begin{document}
  \begin{tikzpicture}

      group style={%

      \addlegendimage{no marks, TolVibBlue!30!black, thick}
