\import{}{pgfplots-def}
\usepgfplotslibrary{groupplots, fillbetween}

\pgfplotscreateplotcyclelist{PointsList}{%
  {TolVibBlue!80!black,   mark=*},
  {TolVibBlue!90!cyan,    mark=*},
  {TolVibCyan,            mark=*},
  {TolVibRed,             mark=*},
  {TolVibOrange,          mark=*},
  {TolBriGreen!70!black,  mark=*},
  {TolBriGreen!90!black,  mark=*},
  {TolBriGreen!90!green,  mark=*},
  {TolBriGreen!70!green,  mark=*},
  {TolBriGreen!50!green,  mark=*},
  {black,                 mark=*},
  {TolBriPurple!50!black, mark=*},
  {TolBriPurple,          mark=*},
}

\pgfplotsset{
  width={4.17cm},
  height=5cm,
  cycle list name=PointsList,
  every axis/.style={
    smooth,
    no marks,
    table/x=par,
  },
  best/.style={%
    forget plot,
    table/y=\point,
  },
  upper/.style={%
    forget plot,
    name path=upper,
    draw=none,
    table/y expr=\thisrow{\point}+\thisrow{\point_err},
  },
  lower/.style={%
    forget plot,
    name path=lower,
    draw=none,
    table/y expr=\thisrow{\point}-\thisrow{\point_err},
  },
  fillreg/.style={%
    opacity=0.2,
  },
}

\def\points{%
  0_30_40,%
  0_30_0,%
  0_30_-30,%
  0_20_0,%
  0_20_-30,%
  0_0_20,%
  0_0_-27,%
  0_0_-40,%
  0_0_-70,%
  0_0_-95,%
  11_0_-24,
  7_0_-5,
  -7_0_6%
}

\begin{document}
  \begin{tikzpicture}
    \begin{groupplot}[
      group style={%
        group size=4 by 1,
        y descriptions at=all,
      },
    ]

    \nextgroupplot[xlabel={Ge reflectivity}, ylabel={probability}]
    \foreach \point in \points {%
      \addplot+[best] table {data/refl-dist-pcanorm.csv};

      \addplot+[fillreg] fill between[of=upper and lower];
    }

    \nextgroupplot[xlabel={LAr abs.~length (cm)}]

      \addplot+[lower] table {data/tpb-dist-pcanorm.csv};

    }

  \end{groupplot}
