    mbb,

    table/y expr={(\thisrow{mbb_90_limit_up}+\thisrow{mbb_90_limit_low})/2},

  exposure/.style={%
    symbols,
    % densely dashed,
    mark=*,
    mark options={solid},
    table/x=date,
    table/y=exposure_kgyr,
  },
